\section{Problem}
Let $f:[-1,1]\rightarrow \vec{R}$ be a function given by
\begin{align*}
%       P_{\theta}(\theta)= \lcbrak{\frac{1}{2\pi}; \ 0\leq \theta \leq 2\pi \\
%                                   0 ; otherwise}
	f(x) =
        \begin{cases}
		x^2cos(\frac{1}{x}) \ x\neq 0 \\
                                   0 \ x=0
        \end{cases}
\end{align*}
Then
\begin{enumerate}
	\item f is of bounded variation on [-1,1]
	\item f' is of bounded variation on [-1,1]
	\item $\abs{f'(x)} \leq$ 1 $\forall x \in [-1,1]$
	\item $\abs{f'(x)} \leq$ 3 $\forall x \in [-1,1]$
\end{enumerate}
\section{Explanation}
We know that
\begin{align}
	-1\leq cos(\frac{1}{x}) \leq 1 \\
	\implies -x^2 \leq x^2cos(\frac{1}{x}) \leq x^2  \label{eq1}
\end{align}
But since $ x \in [-1,1]$
\begin{align}
	-1 \leq x^2cos(\frac{1}{x}) \leq 1
\end{align}
Upon differentiating (\ref{eq1}) we get
\begin{align}
	-2x\leq f' \leq 2x \\
	\implies -2 \leq f' \leq 2\\
	\implies \abs{f'}\leq 2 \quad \forall x \in [-1,1]
\end{align}
Thus f is of bounded variation on [-1,1].
